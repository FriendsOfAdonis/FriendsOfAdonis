---
title: Action as listener
---

Your Actions can be used as [Event Listeners](https://docs.adonisjs.com/guides/events/listeners) that respond to events emitted throughout your application.

## Implementing `AsListener`

By implementing `AsListener` you can define how your action is executed when a specific event is emitted. The interface accepts a generic type parameter to type the event payload.

```ts title="app/actions/send_welcome_email_action.ts" twoslash
// @noErrors
import { BaseAction, AsListener } from '@foadonis/actions'
import User from '#models/user'
import UserRegisteredEvent from '#events/user_registered'
import mail from '@adonisjs/mail/services/main'

export default class SendWelcomeEmailAction extends BaseAction implements AsListener {
  async handle(user: User) {
    await mail.send((message) => {
      message
        .to(user.email)
        .from('hello@friendsofadonis.com')
        .subject('Welcome to Friends of Adonis')
        .htmlView('emails/welcome_email', { user })
    })
  }

  async asListener(event: UserRegisteredEvent) {
    await this.handle(event.user)
  }
}
```

## Registering the listener

You can register your action as an event listener in your `start/events.ts` file:

```ts title="start/events.ts" twoslash
// @noErrors
import emitter from '@adonisjs/core/services/emitter'
import actions from '#generated/actions'
import UserRegisteredEvent from '#events/user_registered'

emitter.on(UserRegisteredEvent, actions.SendWelcomeEmail.asListener())
```
