---
title: renderToReadableStream
description: Render JSX to a Readable Web Stream
---

The `renderToReadableStream` is used to transform Spark tree into a Readable Web Stream.

```tsx twoslash
// @noErrors
import { renderToReadableStream } from '@foadonis/spark/jsx-runtime'

router.get('/', ({ response }: HttpContext) => {
  const readable = renderToReadableStream(<div>Hello world</div>)

  response.stream(Readable.fromWeb(readable))
})
```
