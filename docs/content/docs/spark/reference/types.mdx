### `ComponentProps<T>`

Returns the props of a component.
It can be used with any component, including HTML elements, function components and class components.

```tsx tab="HTML elements"
const Button = (props: ComponentProps<'button'>) => {
  return <button {...props} />
}
```

```tsx tab="Function components"
const Button = ({ label, ...disabled }: { disabled?: boolean; label: string }) => {
  return <button>{label}</button>
}

const PlayButton = ({ label = 'Play', ...props }: ComponentProps<typeof Button>) => {
  return <button label={label} {...props} />
}
```

```tsx tab="Class components"
class Button extends Component<{ label: string }> {
  render() {
    return <button>{this.props.label}</button>
  }
}

const PlayButton = ({ label = 'Play', ...props }: ComponentProps<typeof Button>) => {
  return <button label={label} {...props} />
}
```

### `SparkNode`

Similar to ReactNode, SparkNode represents anything that can be rendered by Spark.
It can be a component, a string, a number, a boolean, an array or a fragment, etc.

```tsx
export const Button = ({ children }: { children: SparkNode }) => {
  return <button>{children}</button>
}
```
